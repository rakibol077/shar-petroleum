/** * ১. ডাটা রিড করার ফাংশন (GET Request) */ function doGet() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const data = sheet.getDataRange().getValues(); // ১ নম্বর রো এখন হেডার (date, station Name, mada, visa, master) const headers = data[0]; const rows = data.slice(1); const result = rows.map((row, index) => { let obj = { rowId: index + 2 }; // শিটের রো নম্বর ট্র্যাক করার জন্য headers.forEach((header, i) => { // কলামের নাম থেকে স্পেস সরিয়ে এবং ছোট হাতের অক্ষরে কি (Key) তৈরি করবে // উদাহরণ: "station Name" হয়ে যাবে "stationname" let key = header.toString().toLowerCase().replace(/\s/g, ''); obj[key] = row[i]; }); return obj; }); return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } /** * ২. ডাটা রাইট, এডিট এবং ডিলিট করার ফাংশন (POST Request) */ function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); let params; try { params = JSON.parse(e.postData.contents); } catch (err) { return ContentService.createTextOutput("Error: Invalid JSON").setMimeType(ContentService.MimeType.TEXT); } const action = params.action; // নতুন ডাটা যোগ করা if (action === "add") { sheet.appendRow([ params.date, params.stationName, params.mada, params.visa, params.master ]); } // বিদ্যমান ডাটা এডিট করা else if (action === "edit") { const rowNum = parseInt(params.index) + 2; sheet.getRange(rowNum, 1, 1, 5).setValues([[ params.date, params.stationName, params.mada, params.visa, params.master ]]); } // ডাটা ডিলিট করা else if (action === "delete") { const rowNum = parseInt(params.index) + 2; sheet.deleteRow(rowNum); } return ContentService.createTextOutput("Success") .setMimeType(ContentService.MimeType.TEXT); }
